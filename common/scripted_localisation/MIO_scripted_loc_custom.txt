define_text = {
	name = RUS_UnlockMIOFocusTrigger
	# THIS = country scope
	# PREV = MIO scope
	text = {
		#RSA always only requires this focus to get the decisions to unlock the MIOs
		# UZTM, CHTZ, TSTZ
		trigger = { original_tag = RSA }
		localization_key = RUS_secret_weapons_development
	}
	text = {
		# UZTM, CHTZ
		trigger = {
			original_tag = FER
			PREV = { NOT = { is_military_industrial_organization = RUS_tstz_organisation } }
		}
		localization_key = RUS_mio_FER_complete_decision_in_foci_tooltip #Both Siberian and Russian FER can get these MIOs
	}
	text = {
		# TSTZ
		trigger = { original_tag = FER }
		localization_key = FER_return_to_kremlin #Only Russian FER can get this MIO
	}
	# RUS is assumed country scope from this point onwards
	text = {
		# UZTM, CHTZ
		trigger = {
			PREV = {
				OR = {
					is_military_industrial_organization = RUS_uztm_organisation
					is_military_industrial_organization = RUS_chtz_organisation
				}
			}
		}
		localization_key = RUS_industrialise_urals
	}
	text = {
		#TSTZ
		trigger = {
			PREV = { is_military_industrial_organization = RUS_tstz_organisation }
		}
		localization_key = RUS_industrialise_eastern_europe
	}
}
