	AUS_reform_faction = {
		icon = generic_political_discourse
		allowed = { tag = AUS }
		
		available = {
			has_war = no
		}
		
		visible = {
			has_completed_focus = AUS_protector_of_the_balkans
			is_subject = no
			has_socialist_government = no
			OR = {
				is_in_faction = no
				is_in_faction_with = GER
			}
			has_country_flag = AUS_intervened_in_weltkrieg
			NOT = { has_global_flag = AUS_treaty_of_budapest }
		}
		
		cost = 0
		fire_only_once = yes

		ai_will_do = {
			factor = 100
		}
		
		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision AUS_reform_faction"
			clr_country_flag = AUS_intervened_in_weltkrieg
			if = {
				limit = { is_in_faction_with = GER }
				leave_faction = yes
				GER = { country_event = aus.tree.4015 }
			}
			create_faction_from_template = { 
				template = faction_template_generic_dominance              
				name = austrian_sphere
				icon = GFX_faction_austria_hungary
			}			

			every_country = {
				limit = {
					has_country_flag = AUS_former_faction_member
					has_socialist_government = no
					OR = {
						is_in_faction_with = GER
						is_in_faction = no
					}
					OR = {
						is_subject_of = ROOT
						is_subject = no
					}
				}
				country_event = aus.tree.4014
			}
		}
	}
