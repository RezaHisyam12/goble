japan_hates_china = {
	reversed = yes
	allowed = {
		is_chinese_tag = yes
	}
	enable_reverse = {
		JAP = { is_subject = no }
	}
	enable = {
		OR = {
			is_neighbor_of = JAP
			tag = FNG
			tag = event_target:current_china_leader
			AND = {
				is_neighbor_of = FNG
				FNG_JAP_hostile = no
			}
		}
		NOT = {
			is_working_with_JAP = yes
			is_in_faction_with = JAP
			is_subject_of = JAP
		}
	}
	target_array = "global.china_tags_array"

	abort_when_not_enabled = yes

	ai_strategy = {
		type = conquer
		id = "JAP"
		value = 100
	}
	ai_strategy = {
		type = antagonize
		id = "JAP"
		value = 1000
	}
}
