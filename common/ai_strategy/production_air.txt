####################################################################################################

light_aircraft_production_default = {
	enable = {
		is_eligible_for_light_aircraft_production = yes
	}
	abort_when_not_enabled = yes

	ai_strategy = { type = air_factory_balance value = 100 }
	ai_strategy = { type = unit_ratio id = fighter value = 70 }
	ai_strategy = { type = unit_ratio id = cas value = 30 }
}

light_aircraft_production_air_war_ongoing = {
	enable = {
		is_eligible_for_light_aircraft_production = yes
		date > 1939.1.1
		has_war = yes
		any_enemy_country = { is_major = yes }
		OR = {
			has_yellow_air = yes
			has_red_air = yes
		}
	}
	abort_when_not_enabled = yes

	ai_strategy = { type = unit_ratio id = fighter value = 10 }
	ai_strategy = { type = unit_ratio id = cas value = -10 }

	ai_strategy = {
		type = equipment_production_factor
		id = fighter
		value = 25
	}
	ai_strategy = {
		type = equipment_production_factor
		id = cas
		value = -25
	}
	ai_strategy = {
		type = equipment_production_factor
		id = tactical_bomber
		value = -25
	}
	ai_strategy = {
		type = equipment_production_factor
		id = strategic_bomber
		value = -25
	}
}
