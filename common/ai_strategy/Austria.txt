austria_ignore_romania = {
	allowed = {
		OR = {
			original_tag = AUS
			is_in_faction_with = AUS
			is_subject_of = AUS
		}
	}
	enable = {
		NOT = {
			ROM = {
				is_justifying_wargoal_against = AUS
			}
			AND = {
				ROM = {
					is_in_faction_with = SER
				}
				SER = {
					is_justifying_wargoal_against = AUS
				}
			}
		}
	}

	abort_when_not_enabled = yes

	ai_strategy = {
		type = ignore
		id = ROM
		value = 500
	}
}

	}
}
