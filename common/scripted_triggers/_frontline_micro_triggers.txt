has_selected_a_grand_doctrine = {
	OR = {
		has_doctrine = new_mobile_warfare
		has_doctrine = superior_firepower
		has_doctrine = grand_battleplan
		has_doctrine = mass_assault
	}
}

has_one_subdoctrine_track = {
	has_subdoctrine_in_track = infantry
}

has_two_subdoctrine_tracks = {
	has_subdoctrine_in_track = infantry
	OR = {
		has_subdoctrine_in_track = combat_support
		has_subdoctrine_in_track = operations
	}
}

has_three_subdoctrine_tracks = {
	has_subdoctrine_in_track = infantry
	has_subdoctrine_in_track = combat_support
	has_subdoctrine_in_track = operations
}

has_four_subdoctrine_tracks = {
	has_subdoctrine_in_track = infantry
	has_subdoctrine_in_track = combat_support
	has_subdoctrine_in_track = armor
	has_subdoctrine_in_track = operations
}

is_losing_tank_war = {
	any_enemy_country = {
		ROOT = {
			estimated_intel_max_armor = {
				tag = PREV
				value > 25
			}
		}
	}
}

has_green_air = {
	has_country_flag = FAI_strong_green_air
}

has_yellow_air = {
	NOT = { has_country_flag = FAI_strong_green_air }
	has_country_flag = FAI_weak_green_air
}

has_red_air = {
	NOT = { has_country_flag = FAI_weak_green_air }
}

the_air_force_is_dead_jim = {
	has_deployed_air_force_size = { type = fighter	size < 100 }
}

has_manpower_and_equipment_surplus = {
	manpower_per_military_factory > 5000
	stockpile_ratio = { archetype = infantry_equipment ratio > 0.05 }
	stockpile_ratio = { archetype = support_equipment ratio > 0.05 }
	stockpile_ratio = { archetype = artillery_equipment ratio > 0.05 }
	stockpile_ratio = { archetype = anti_air_equipment ratio > 0.05 }
	stockpile_ratio = { archetype = anti_tank_equipment ratio > 0.05 }
	stockpile_ratio = { archetype = motorized_equipment ratio > 0.05 }
}

has_manpower_or_equipment_shortage = {
	OR = {
		manpower_per_military_factory < 1000
		stockpile_ratio = { archetype = infantry_equipment ratio < 0.01 }
		stockpile_ratio = { archetype = support_equipment ratio < 0.01 }
		stockpile_ratio = { archetype = artillery_equipment ratio < 0.01 }
		stockpile_ratio = { archetype = anti_air_equipment ratio < 0.01 }
		stockpile_ratio = { archetype = anti_tank_equipment ratio < 0.01 }
		stockpile_ratio = { archetype = motorized_equipment ratio < 0.01 }
	}
}

has_manpower_or_equipment_crisis = {
	OR = {
		has_country_flag = FAI_manpower_crisis
		has_country_flag = FAI_equipment_crisis
	}
}

has_an_actual_civil_war = {
	has_civil_war = yes
	NOT = {
		any_enemy_country = {
			has_civil_war = no
		}
	}
}

has_2acw = {
	is_american_tag = yes
	has_global_flag = USA_civil_war_begun_flag
	NOT = { has_global_flag = USA_civil_war_over_flag }
}

is_civil_warring = {
	OR = {
		has_an_actual_civil_war = yes
		has_2acw = yes
	}
}
