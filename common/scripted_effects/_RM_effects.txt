GRU_add_to_zollverein = {
	if = {
		limit = {
			NOT = { has_country_flag = GRU_zollverein_member }
		}
		set_country_flag = GRU_zollverein_member
		add_to_array = { global.zollverein_members = THIS }
		if = {
			limit = { NOT = { has_dynamic_modifier = { modifier = GRU_zollverein_economic_sphere_member } } }
			add_dynamic_modifier = { modifier = GRU_zollverein_economic_sphere_member }
		}
		if = {
			limit = { NOT = { is_in_tech_sharing_group = zollverein_research } }
			add_to_tech_sharing_group = zollverein_research
		}
		GRU_recalculate_economic_integration = yes
	}
}

		clr_country_flag = GRU_zollverein_member
		remove_from_array = { global.zollverein_members = THIS }
	}
	if = {
		limit = { is_in_tech_sharing_group = zollverein_research }
		remove_from_tech_sharing_group = zollverein_research
	}
}

GRU_recalculate_economic_integration = {
	hidden_effect = {
		set_variable = { global.var_zollverein_factories = 0 }
		for_each_scope_loop = {
			array = global.zollverein_members
			add_to_variable = { global.var_zollverein_factories = THIS.num_of_civilian_factories }
		}

		#German's building boost
		set_variable = { global.zollverein_slots_boost = global.var_zollverein_factories }
		clamp_variable = { var = global.zollverein_slots_boost max = 10 min = 0 }
		multiply_variable = { global.zollverein_slots_boost = 0.75 }
		multiply_variable = { global.zollverein_slots_boost = 0.01 }

		#German's industry boost
		set_variable = { global.zollverein_industry_boost = global.var_zollverein_factories }
		clamp_variable = { var = global.zollverein_industry_boost max = 10 min = 0 }
		multiply_variable = { global.zollverein_industry_boost = 0.01 }
		multiply_variable = { global.zollverein_industry_boost = 0.50 }

		#German's consumer research boost
		set_variable = { global.zollverein_goods_boost = global.var_zollverein_factories }
		clamp_variable = { var = global.zollverein_goods_boost max = 10 min = 0 }
		multiply_variable = { global.zollverein_goods_boost = 0.01 }

		#Member's consumer research boost
		set_variable = { global.zollverein_goods_boost_member = global.var_zollverein_factories }
		add_to_variable = { global.zollverein_goods_boost_member = GRU.num_of_civilian_factories }
		clamp_variable = { var = global.zollverein_goods_boost_member max = 10 min = 0 }
		multiply_variable = { global.zollverein_goods_boost_member = 0.0005 }

		#Member's resources boost
		set_variable = { global.zollverein_resources_boost_member = global.var_zollverein_factories }
		add_to_variable = { global.zollverein_resources_boost_member = GRU.num_of_civilian_factories }
		clamp_variable = { var = global.zollverein_resources_boost_member max = 10 min = 0 }
		multiply_variable = { global.zollverein_resources_boost_member = 0.01 }
		multiply_variable = { global.zollverein_resources_boost_member = 0.50 }
	}
}

GRU_zollverein_effect = {
	if = {
		limit = { NOT = { tag = ROOT } }
		custom_effect_tooltip = if_they_accept_tt
	}
	add_opinion_modifier = {
		target = GRU
		modifier = KR_approves_of_govt
	}
	if = {
		limit = {
			tag = CEN
			OR = {
				312 = { #Honduras
					is_owned_and_controlled_by = PREV
					NOT = { has_state_flag = GRU_zollverein_developed }
				}
				313 = { #Guatemala
					is_owned_and_controlled_by = PREV
					NOT = { has_state_flag = GRU_zollverein_developed }
				}
				314 = { #El Salvador
					is_owned_and_controlled_by = PREV
					NOT = { has_state_flag = GRU_zollverein_developed }
				}
				316 = { #Costa Rica
					is_owned_and_controlled_by = PREV
					NOT = { has_state_flag = GRU_zollverein_developed }
				}
				317 = { #Managua
					is_owned_and_controlled_by = PREV
					NOT = { has_state_flag = GRU_zollverein_developed }
				}
			}
		}
		every_owned_state = {
			limit = {
				OR = {
					state = 312 #Honduras
					state = 313 #Guatemala
					state = 314 #El Salvador
					state = 316 #Costa Rica
					state = 317 #Managua
				}
				NOT = { has_state_flag = GRU_zollverein_developed }
			}
			GRU_zollverein_development_in_targeted_state = yes
		}
	}
	else_if = {
		limit = { has_variable = GRU_num_target_states }
		GRU_zollverein_development_in_random_state = yes
		GRU_zollverein_infrastructure_effect = yes
		if = {
			limit = { check_variable = { GRU_num_target_states > 1 } }
			GRU_zollverein_development_in_random_state = yes
			GRU_zollverein_infrastructure_effect = yes
		}
		if = {
			limit = { check_variable = { GRU_num_target_states > 2 } }
			GRU_zollverein_development_in_random_state = yes
			GRU_zollverein_infrastructure_effect = yes
		}
		if = {
			limit = { check_variable = { GRU_num_target_states > 3 } }
			GRU_zollverein_development_in_random_state = yes
			GRU_zollverein_infrastructure_effect = yes
		}
	}
	else = {
		GRU_zollverein_development_in_capital = yes
		GRU_zollverein_infrastructure_effect = yes
	}
	GRU_add_to_zollverein = yes
	if = {
		limit = { NOT = { tag = HUN } }
		add_tech_bonus = {
			name = GRU_Develop_Country
			bonus = 1
			uses = 1
			category = industry
		}
	}
}

GRU_zollverein_development_in_capital = {
	if = {
		limit = {
			capital_scope = {
				free_building_slots = { building = industrial_complex size > 0 include_locked = yes }
			}
		}
		capital_scope = {
			add_extra_state_shared_building_slots = 1
			add_building_construction = {
				type = industrial_complex
				level = 1
				instant_build = yes
			}
		}
	}
	else = {
		GRU_zollverein_development_in_random_state = yes
	}
}

GRU_zollverein_development_in_targeted_state = {
	if = {
		limit = {
			free_building_slots = { building = industrial_complex size > 0 include_locked = yes }
			NOT = { has_state_flag = GRU_zollverein_developed }
		}
		set_state_flag = GRU_zollverein_developed
		add_extra_state_shared_building_slots = 1
		add_building_construction = {
			type = infrastructure
			level = 1
			instant_build = yes
		}
		add_building_construction = {
			type = industrial_complex
			level = 1
			instant_build = yes
		}
	}
	else = {
		GRU_zollverein_development_in_random_state = yes
		GRU_zollverein_infrastructure_effect = yes
	}
}

		}
	}
}
